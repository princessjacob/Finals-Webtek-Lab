REQUEST (req_id, reqStatus, reqDets, reqDate, cust_id, sp_id)
	FK 	cust_id REFERENCES CUSTOMER Nulls Not Allowed
			Delete Restrict, Update Cascade
		sp_id REFERENCES SERVICE_PROVIDER Nulls Not Allowed
			Delete Restrict, Update Cascade

	<!-- Yung reqdate is para sa transactions part. diba pwedeng magcancel yung customer given na less

SSP (serv_id, sp_id, Price, Rate)
	FK	serv_id REFERENCES SERVICES Nulls Not Allowed
			Delete Restrict, Update Cascade
		sp_id REFERENCES SERVICE_PROVIDER Nulls Not Allowed
			Delete Restrict, Update Cascade

	<!-- Di ko alam kung yung SSP eh tables tungkol sa SP at yung mga services na prinoprovide nila. Pero

		price na iproprovide ng SP. Kunwari diba kapag training eh pwedeng basic training na "sit". So parang
		mas mababa yung price nun kesa sa training na "role over". Tapos nakadepende narin sa SP yung price
